-- Table structure for table <student>

create table student(rollno char(5) primary key, sname varchar(15) not null, address varchar(15) not null, age int not null);
-- Table structure for table <faculty>

create table faculty(fno char(5) primary key, fname varchar(15) not null, specialization varchar(15));

-- Table structure for table <course>
create table course(cno char(5)primary key,cname varchar(15) not null,sem int not null, fees float(5,2)not null, fno char(5) foreign key(fno) references faculty(fno));

-- Table structure for table <enroll>
create table enroll(rollno char(5),cno char(5),edate date not null, primary key(rollno,cno),foreign key(rollno)references student(rollno),foreign key(cno) references course(cno));
